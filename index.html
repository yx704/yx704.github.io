async function sendMessage() { const text = userInput.value.trim(); if (!text) return; addMessage(text, true); userInput.value = ""; sendBtn.disabled = true; const thinking = document.createElement("div"); thinking.className = "message ai-message"; thinking.innerText = "思考中..."; chatMessages.appendChild(thinking); chatMessages.scrollTop = chatMessages.scrollHeight; try { // ============================================== // 🔥 这里是 100% 不会出错的请求体 // ============================================== const bodyToSend = { "inputs": {}, "query": text, "response_mode": "streaming", "user": "frontend-user" }; if (conversationId) { bodyToSend.conversation_id = conversationId; } const res = await fetch(API_URL, { method: "POST", headers: { "Authorization": "Bearer " + API_KEY, "Content-Type": "application/json" }, body: JSON.stringify(bodyToSend) }); const reader = res.body.getReader(); const decoder = new TextDecoder(); let fullAnswer = ""; while (true) { const { done, value } = await reader.read(); if (done) break; const chunk = decoder.decode(value); const lines = chunk.split("\n").filter(l => l.trim()); for (const line of lines) { if (line.startsWith("data:")) { try { const data = JSON.parse(line.slice(5)); conversationId = data.conversation_id; if (data.event === "human_input_required") { chatMessages.removeChild(thinking); addMessage(data.resolved_default_values.user_input, false); sendBtn.disabled = false; return; } if (data.event === "message" && data.answer) { fullAnswer += data.answer; thinking.innerText = fullAnswer; } } catch (e) {} } } } chatMessages.removeChild(thinking); addMessage(fullAnswer || "完成", false); } catch (err) { thinking.innerText = "出错：" + err.message; console.error(err); } finally { sendBtn.disabled = false; } }
